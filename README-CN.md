## 安装

```
npm install vue-easytable
```

or

```
yarn add vue-easytable
```

## 使用

Write the following in main.js:

```javascript
import Vue from "vue";
import "vue-easytable/libs/theme-default/index.css";
import VueEasytable from "vue-easytable";

Vue.use(VueEasytable);

new Vue({
    el: "#app",
    render: (h) => h(App),
});
```
